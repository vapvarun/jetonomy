```php
public function prune_activity_log(): void {

    $table = table( 'activity_log' );
```

```php
// jetonomy-pro.php - refactor the existing register_activation_hook() at line 63

register_activation_hook( __FILE__, __NAMESPACE__ . '\\activate_plugin' );

function activate_plugin( bool $network_wide = false ): void {
    if ( is_multisite() && $network_wide ) {
        $blog_ids = get_sites( array( 'fields' => 'ids', 'number' => 0 ) );
        foreach ( $blog_ids as $blog_id ) {
            switch_to_blog( (int) $blog_id );
            do_pro_install();
            restore_current_blog();
        }
        return;
    }
    do_pro_install();
}

function do_pro_install(): void {
    // Iterate every loaded extension's activate(); guarded so missing
    // extensions or partial Pro_Schema state can't fatal the install.
    foreach ( \Jetonomy_Pro::instance()->extensions() as $extension ) {
        if ( method_exists( $extension, 'activate' ) ) {
            $extension->activate();
        }
    }
}

add_action( 'wp_initialize_site', __NAMESPACE__ . '\\install_on_new_site', 10, 2 );

function install_on_new_site( $new_site, array $args ): void {
    if ( ! is_a( $new_site, 'WP_Site' ) ) {
        return;
    }
    switch_to_blog( (int) $new_site->blog_id );
    do_pro_install();
    restore_current_blog();
}
```
